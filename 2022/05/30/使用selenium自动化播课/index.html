def timeToS(timeStr):
    t = timeStr.strip().split(':')
    if len(t) == 2:
        mm, ss = t
        return int(mm) * 60 + int(ss)
    elif len(t) == 3:
        hh, mm, ss = t
        return int(hh) * 3600 + int(mm) * 60 + int(ss)
